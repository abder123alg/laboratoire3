@startuml

title Creation du labyrinthe

participant "jeu:Jeu" as jeu
participant "laby:zeldiablo.Labyrinthe" as laby
participant "c:zeldiablo.Case" as c

create jeu
create laby
activate jeu
jeu -> laby : new zeldiablo.Labyrinthe()

loop
    activate laby
    create c
    laby -> c : new zeldiablo.Case()
end

deactivate laby
deactivate jeu

@enduml
