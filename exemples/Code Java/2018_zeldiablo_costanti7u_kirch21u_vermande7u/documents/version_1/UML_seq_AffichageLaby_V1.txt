@startuml

title Affichage du labyrinthe

participant "zeldiablo.Main" as main
participant "j:Jeu" as jeu
participant "laby:zeldiablo.Labyrinthe" as laby
participant "p:zeldiablo.Personnage" as perso

activate main
activate jeu
    main -> jeu : toString()
    activate laby
    jeu -> laby : toString()
    laby --> jeu : String()
    deactivate laby
    activate perso
    jeu -> perso : getX()
    perso --> jeu : int
    jeu -> perso : getY()
    perso --> jeu : int
    deactivate perso


    jeu --> main : String
deactivate jeu
deactivate main


@enduml

test!!
