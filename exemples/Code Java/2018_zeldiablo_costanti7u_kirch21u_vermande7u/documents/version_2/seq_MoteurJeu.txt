@startuml
title zeldiablo.Personnage se deplace

participant main
participant "mg:MoteurGraphique" as mg
participant "jeu:zeldiablo.JeuZeldiablo" as j
participant "p:zeldiablo.Personnage" as p
participant "dessin:zeldiablo.DessinZeldiablo" as d


activate main
create j
main -> j : new
create p
j -> p : new
create d
main -> d : new (jeu)
create mg
main -> mg : new (jeu, dessin)


main->mg : lancerJeu(int w, int h)
activate mg

loop non jeu.etreFini()

   mg->j : evoluer(Commande c)
   activate j

   j->p : seDeplacer(zeldiablo.Direction d)
   activate p

   p-->j
   deactivate p

   j-->mg
   deactivate j

   mg->d : dessiner(BufferedImage bf)
   activate d
   d-->mg
   deactivate d
end

mg-->main
deactivate mg
deactivate main
@enduml
